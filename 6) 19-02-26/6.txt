Assignment: Student Data Manager

Date: 19/02/2026

Description

This Python program stores the marks of 5 students using dictionaries. It then:

Finds the topper (highest marks)
Calculates the class average
Assigns grades to each student based on their marks
Python Code
# Student Data Manager

students = {}

# Input data for 5 students
for i in range(5):
    name = input("Enter student name: ")
    marks = float(input("Enter marks: "))
    students[name] = marks

# Find topper
topper = max(students, key=students.get)

# Calculate class average
average = sum(students.values()) / len(students)

# Function to assign grades
def get_grade(marks):
    if marks >= 90:
        return "A"
    elif marks >= 75:
        return "B"
    elif marks >= 60:
        return "C"
    elif marks >= 40:
        return "D"
    else:
        return "F"

# Print student grades
print("\nStudent Grades:")
for name, marks in students.items():
    print(name, "-", marks, "Grade:", get_grade(marks))

# Print topper
print("\nTopper:", topper, "with", students[topper], "marks")

# Print class average
print("Class Average:", average)
